91  systemData.buttons[BUTTON_HIGH] = 0; // Marker for last button.
92 
93  // Request values for all buttons in turn.
94  for (int button = 0; button < BUTTON_HIGH; button++)
95  {
96  disp.clearRow(LCD_COL_CALIBRATE, LCD_ROW_DET);
97  disp.printProgStrAt(LCD_COL_CALIBRATE, LCD_ROW_DET, M_BUTTONS[button + 1], LCD_LEN_OPTION);
98 
99  // Wait for a button to be pressed
100  while ((value = analogRead(A0)) > BUTTON_THRESHHOLD)
101  {
102  delay(DELAY_BUTTON_WAIT);
103  }
104 
105  if (isDebug(DEBUG_BRIEF))
106  {
107  Serial.print(millis());
108  Serial.print(CHAR_TAB);
109  Serial.print(PGMT(M_BUTTONS[button + 1]));
110  Serial.print(PGMT(M_DEBUG_VALUE));
111  Serial.print(value, HEX);
112  Serial.print(PGMT(M_DEBUG_TARGET));
113  Serial.print(previous, HEX);
114  Serial.println();
115  }
116 
117  // Report the button's value (4 digits)
118  disp.setCursor(LCD_COLS - 4, LCD_ROW_DET);
119  disp.printDec(value, 4, CHAR_SPACE);
120 
121  // Wait for button to be released.
122  while (analogRead(A0) < BUTTON_THRESHHOLD)
123  {
124  delay(DELAY_BUTTON_WAIT);
125  }
126

128  if (previous < value)

131  delay(DELAY_READ);
132

136  disp.clearRow(LCD_COL_START, LCD_ROW_DET);
137  disp.printProgStrAt(LCD_COL_START, LCD_ROW_DET, M_PRESS, LCD_COLS);
138  }
