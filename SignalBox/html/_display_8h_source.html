340  void printDec(int aValue, int aDigits, char aPad)
341  {
342  int value = aValue;
343  int digits = 1;
344  int divisor = 1;
345  bool leadingBlanks = true;
346 
347  // Calculate min digits required
348  while (value >= 10)
349  {
350  value /= 10;
351  digits += 1;
352  }
353  value = aValue;
354 
355  // Use at least the min digits requested.
356  if (digits < aDigits)
357  {
358  digits = aDigits;
359  }
360 
361  // Calculate starting power of 10 for desired number of digits.
362  for (int digit = 1; digit < digits; digit++)
363  {
364  divisor *= 10;
365  }
366 
367  // Output the digits in sequence.
368  while (--digits >= 0)
369  {
370  if (value / divisor > 0)
371  {
372  printCh((char) (CHAR_ZERO + (value / divisor)));
373  leadingBlanks = false;
374  }
375  else if ( (digits > 0) // Don't pad the last digit
376  && (leadingBlanks)) // Don't pad after we've had some non-zero digits.
377  {
378  printCh(aPad);
379  }
380  else
381  {
382  printCh(CHAR_ZERO);
383  }
384 
385  // Next digit.
386  value = value % divisor;
387  divisor /= 10;
388  }
